**理解new关键字**

```js
// 我们知道实例化一个对象是隐式原型，指向构造函数的显示原型
obj1.__proto__ = OBJ.prototype

function OBJ() {
  // 需要一个空的对象
  var obj = new Object()
  // 让这个obj 指向构造函数原型
  obj.__proto__ = OBJ.prototype
  // 改变this的指向
  OBJ.call(obj)
}
```
